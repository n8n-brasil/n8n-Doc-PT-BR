## <IonicIcon name="analytics-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Métricas Essenciais

### Métricas de Sistema

#### **CPU e Memória**

```bash
#!/bin/bash
# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> system-metrics.sh

echo "=== Métricas do Sistema ==="

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> CPU
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
echo "CPU Usage: ${CPU_USAGE}%"

# <IonicIcon name="sparkles-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Memória
MEMORY_USAGE=$(free | grep Mem | awk '{printf("%.2f", $3/$2 * 100.0)}')
echo "Memory Usage: ${MEMORY_USAGE}%"

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Disco
DISK_USAGE=$(df / | tail -1 | awk '{print $5}' | cut -d'%' -f1)
echo "Disk Usage: ${DISK_USAGE}%"

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Load Average
LOAD_AVG=$(uptime | awk -F'load average:' '{print $2}' | awk '{print $1}' | cut -d',' -f1)
echo "Load Average: $LOAD_AVG"
```

#### **Métricas de Rede**

```bash
#!/bin/bash
# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> network-metrics.sh

echo "=== Métricas de Rede ==="

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Conexões ativas
ACTIVE_CONNECTIONS=$(netstat -an | grep ESTABLISHED | wc -l)
echo "Active Connections: $ACTIVE_CONNECTIONS"

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Conexões n8n
N8N_CONNECTIONS=$(netstat -an | grep :5678 | grep ESTABLISHED | wc -l)
echo "n8n Connections: $N8N_CONNECTIONS"

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Tráfego de rede
NETWORK_IN=$(cat /proc/net/dev | grep eth0 | awk '{print $2}')
NETWORK_OUT=$(cat /proc/net/dev | grep eth0 | awk '{print $10}')
echo "Network In: $NETWORK_IN bytes"
echo "Network Out: $NETWORK_OUT bytes"
```

### Métricas de Aplicação

#### **Métricas n8n**

```bash
#!/bin/bash
# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> n8n-metrics.sh

N8N_URL="http://localhost:5678"
API_KEY="sua_api_key"

echo "=== Métricas n8n ==="

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Status da aplicação
if curl -f -s "$N8N_URL/healthz" > /dev/null; then
    echo "n8n Status: ✅ Online"
else
    echo "n8n Status: ❌ Offline"
    exit 1
fi

# <IonicIcon name="git-branch-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Número de workflows
WORKFLOW_COUNT=$(curl -s -H "X-N8N-API-KEY: $API_KEY" \
  "$N8N_URL/api/v1/workflows" | jq '. | length')
echo "Total Workflows: $WORKFLOW_COUNT"

# <IonicIcon name="git-branch-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Workflows ativos
ACTIVE_WORKFLOWS=$(curl -s -H "X-N8N-API-KEY: $API_KEY" \
  "$N8N_URL/api/v1/workflows" | jq '[.[] | select(.active == true)] | length')
echo "Active Workflows: $ACTIVE_WORKFLOWS"

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Execuções nas últimas 24h
EXECUTIONS_24H=$(curl -s -H "X-N8N-API-KEY: $API_KEY" \
  "$N8N_URL/api/v1/executions?limit=1000" | \
  jq '[.[] | select(.startedAt > "'$(date -d '24 hours ago' -Iseconds)'")] | length')
echo "Executions (24h): $EXECUTIONS_24H"

# <IonicIcon name="bug-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Taxa de erro
ERROR_RATE=$(curl -s -H "X-N8N-API-KEY: $API_KEY" \
  "$N8N_URL/api/v1/executions?limit=1000" | \
  jq '[.[] | select(.startedAt > "'$(date -d '1 hour ago' -Iseconds)'")] | 
       [.[] | select(.status == "error")] | length as $errors | 
       [.[] | select(.startedAt > "'$(date -d '1 hour ago' -Iseconds)'")] | length as $total | 
       if $total > 0 then ($errors * 100 / $total) else 0 end')
echo "Error Rate (1h): ${ERROR_RATE}%"
```

#### **Métricas de Performance**

```bash
#!/bin/bash
# <IonicIcon name="speedometer-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> performance-metrics.sh

echo "=== Métricas de Performance ==="

# <IonicIcon name="code-slash-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Tempo de resposta da API

  http://localhost:5678/api/v1/workflows)

# <IonicIcon name="grid-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Tempo de resposta da UI
UI_RESPONSE_TIME=$(curl -o /dev/null -s -w "%{time_total}" \
  http://localhost:5678/)

NODE_MEMORY=$(docker exec n8n node -e "
const mem = process.memoryUsage();
console.log(Math.round(mem.rss / 1024 / 1024));
" 2>/dev/null || echo "N/A")
echo "Node.js Memory: ${NODE_MEMORY}MB"

# <IonicIcon name="code-slash-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Uso de CPU do Node.js

echo "Node.js CPU: $NODE_CPU"
```

## <IonicIcon name="color-palette-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Sistema de Alertas

### Configuração de Alertas

#### **Script de Monitoramento**

```bash
#!/bin/bash
# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> n8n-monitoring.sh

# <IonicIcon name="key-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Configurações
WEBHOOK_URL="https://hooks.slack.com/services/YOUR/WEBHOOK/URL"
EMAIL_ALERTS="admin@empresa.com"
ALERT_THRESHOLD_CPU=80
ALERT_THRESHOLD_MEMORY=85
ALERT_THRESHOLD_DISK=90
ALERT_THRESHOLD_ERROR_RATE=5
ALERT_THRESHOLD_RESPONSE_TIME=5

# <IonicIcon name="code-slash-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Função para enviar alerta
send_alert() {
    local message="$1"
    local severity="$2"
    
    # Slack
    curl -X POST $WEBHOOK_URL \
      -H "Content-type: application/json" \
      -d "{\"text\":\"$severity $message\"}"
    
    # Email
    echo "$message" | mail -s "n8n Alert: $severity" $EMAIL_ALERTS
    
    # Log
    echo "$(date): $severity - $message" >> /var/log/n8n/alerts.log
}

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar CPU
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
if (( $(echo "$CPU_USAGE > $ALERT_THRESHOLD_CPU" | bc -l) )); then
    send_alert "CPU usage is ${CPU_USAGE}%" "⚠️"
fi

# <IonicIcon name="sparkles-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar memória
MEMORY_USAGE=$(free | grep Mem | awk '{printf("%.0f", $3/$2 * 100.0)}')
if [ $MEMORY_USAGE -gt $ALERT_THRESHOLD_MEMORY ]; then
    send_alert "Memory usage is ${MEMORY_USAGE}%" "⚠️"
fi

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar disco
DISK_USAGE=$(df / | tail -1 | awk '{print $5}' | cut -d'%' -f1)
if [ $DISK_USAGE -gt $ALERT_THRESHOLD_DISK ]; then
    send_alert "Disk usage is ${DISK_USAGE}%" "🚨"
fi

# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar n8n
if ! curl -f -s http://localhost:5678/healthz > /dev/null; then
    send_alert "n8n is not responding" "🚨"
fi

# <IonicIcon name="bug-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar taxa de erro
ERROR_RATE=$(curl -s -H "X-N8N-API-KEY: $API_KEY" \
  "http://localhost:5678/api/v1/executions?limit=1000" | \
  jq '[.[] | select(.startedAt > "'$(date -d '10 minutes ago' -Iseconds)'")] | 
       [.[] | select(.status == "error")] | length as $errors | 
       [.[] | select(.startedAt > "'$(date -d '10 minutes ago' -Iseconds)'")] | length as $total | 
       if $total > 0 then ($errors * 100 / $total) else 0 end')

if (( $(echo "$ERROR_RATE > $ALERT_THRESHOLD_ERROR_RATE" | bc -l) )); then
    send_alert "Error rate is ${ERROR_RATE}%" "🚨"
fi

# <IonicIcon name="time-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar tempo de resposta
RESPONSE_TIME=$(curl -o /dev/null -s -w "%{time_total}" \
  http://localhost:5678/api/v1/workflows)

if (( $(echo "$RESPONSE_TIME > $ALERT_THRESHOLD_RESPONSE_TIME" | bc -l) )); then
    send_alert "Response time is ${RESPONSE_TIME}s" "⚠️"
fi
```

### Alertas Baseados em Tempo

#### **Alertas de Horário**

```bash
#!/bin/bash
# <IonicIcon name="time-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> time-based-alerts.sh

# <IonicIcon name="key-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Configurações
CURRENT_HOUR=$(date +%H)
CURRENT_DAY=$(date +%u)  # 1=Monday, 7=Sunday
BUSINESS_HOURS_START=9
BUSINESS_HOURS_END=18

# <IonicIcon name="time-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> Verificar se está em horário comercial
if [ $CURRENT_DAY -ge 1 ] && [ $CURRENT_DAY -le 5 ]; then
    if [ $CURRENT_HOUR -ge $BUSINESS_HOURS_START ] && [ $CURRENT_HOUR -lt $BUSINESS_HOURS_END ]; then
        # Em horário comercial - alertas mais rigorosos
        ALERT_THRESHOLD_CPU=70
        ALERT_THRESHOLD_MEMORY=80
        ALERT_THRESHOLD_ERROR_RATE=2
    else
        # Fora do horário comercial - alertas mais flexíveis
        ALERT_THRESHOLD_CPU=90
        ALERT_THRESHOLD_MEMORY=90
        ALERT_THRESHOLD_ERROR_RATE=10
    fi
else
    # Fim de semana - alertas muito flexíveis
    ALERT_THRESHOLD_CPU=95
    ALERT_THRESHOLD_MEMORY=95
    ALERT_THRESHOLD_ERROR_RATE=20
fi

echo "Current thresholds - CPU: ${ALERT_THRESHOLD_CPU}%, Memory: ${ALERT_THRESHOLD_MEMORY}%, Error Rate: ${ALERT_THRESHOLD_ERROR_RATE}%"
```

#### **Custom Metrics**

```javascript
// n8n-datadog-metrics.js
const { StatsD } = require('hot-shots');

const dogstatsd = new StatsD({
  host: 'localhost',
  port: 8125,
  prefix: 'n8n.'
});

// Métricas de workflow
function trackWorkflowExecution(workflowId, duration, status) {
  dogstatsd.timing('workflow.execution_time', duration, [`workflow_id:${workflowId}`]);
  dogstatsd.increment('workflow.executions', 1, [`status:${status}`, `workflow_id:${workflowId}`]);
}

// Métricas de erro
function trackError(errorType, errorMessage) {
  dogstatsd.increment('errors.total', 1, [`type:${errorType}`]);
  dogstatsd.event('n8n.error', errorMessage, {
    alert_type: 'error',
    tags: [`error_type:${errorType}`]
  });
}

// Métricas de performance
function trackAPIPerformance(endpoint, duration, statusCode) {
  dogstatsd.timing('api.response_time', duration, [`endpoint:${endpoint}`]);
  dogstatsd.increment('api.requests', 1, [`endpoint:${endpoint}`, `status:${statusCode}`]);
}
```

#### **Logstash Pipeline**

```ruby
# <IonicIcon name="document-outline" style={{ fontSize: '24px', color: '#ea4b71' }} /> logstash/pipeline/n8n.conf
input {
  beats {
    port => 5044
  }
}

filter {
  if [fields][service] == "n8n" {
    json {
      source => "message"
    }
    
    date {
      match => [ "timestamp", "ISO8601" ]
      target => "@timestamp"
    }
    
    mutate {
      add_field => {
        "service" => "n8n"
        "environment" => "%{ENVIRONMENT}"
      }
    }
  }
}

output {
  elasticsearch {
    hosts => ["elasticsearch:9200"]
    index => "n8n-logs-%{+YYYY.MM.dd}"
  }
}
```
