# AWS Brasil - Deploy SUA_SENHA_BANCO_AQUI na Nuvem

Este guia mostra como fazer deploy do SUA_SENHA_BANCO_AQUI na AWS Brasil, aproveitando a infraestrutura local para melhor performance e compliance com regulamentações brasileiras.

## <IonicIcon name="settings-outline" style={{fontSize: '24px', color: '#ea4b71'}} /> Configuração de Recursos AWS

### Terraform Configuration

```hcl
# main.tf
provider "aws" {
  region = "us-east-1"
}

# VPC Resource
resource "aws_vpc" "SUA_SENHA_BANCO_AQUI_vpc" {
  cidr_block = "10.0.0.0/16"
  
  tags = {
    Name = "SUA_SENHA_BANCO_AQUI-vpc"
    Environment = "production"
    Project = "SUA_SENHA_BANCO_AQUI-automation"
  }
}

# EC2 Instance
resource "aws_instance" "SUA_SENHA_BANCO_AQUI_server" {
  ami           = "ami-0c02fb55956c7d316" # Amazon Linux 2
  instance_type = "t3.medium"
  
  vpc_security_group_ids = [aws_security_group.SUA_SENHA_BANCO_AQUI_sg.id]
  subnet_id              = aws_subnet.SUA_SENHA_BANCO_AQUI_public.id
  
  user_data = <<-EOF
              #!/bin/bash
              yum update -y
              yum install -y docker
              systemctl start docker
              systemctl enable docker
              docker run -d --name SUA_SENHA_BANCO_AQUI -p 5678:5678 SUA_SENHA_BANCO_AQUIio/SUA_SENHA_BANCO_AQUI:latest
              EOF
  
  tags = {
    Name = "SUA_SENHA_BANCO_AQUI-server"
  }
}

# RDS PostgreSQL
resource "aws_db_instance" "SUA_SENHA_BANCO_AQUI_database" {
  identifier = "SUA_SENHA_BANCO_AQUI-db"
  
  engine         = "postgres"
  engine_version = "13.7"
  instance_class = "db.t3.micro"
  
  allocated_storage     = 20
  storage_type         = "gp2"
  storage_encrypted    = true
  
  db_name  = "SUA_SENHA_BANCO_AQUI"
  username = "SUA_SENHA_BANCO_AQUI_user"
  password = var.db_password
  
  vpc_security_group_ids = [aws_security_group.rds_sg.id]
  db_subnet_group_name   = aws_db_subnet_group.SUA_SENHA_BANCO_AQUI_db_subnet.name
  
  backup_retention_period = 7
  backup_window          = "03:00-04:00"
  maintenance_window     = "sun:04:00-sun:05:00"
  
  tags = {
    Name = "SUA_SENHA_BANCO_AQUI-database"
  }
}
```

## <IonicIcon name="warning-outline" style={{fontSize: '24px', color: '#ea4b71'}} /> Monitoramento e Alertas

### CloudWatch Métricas Essenciais

```bash
# Configurar alarme de CPU
aws cloudwatch put-metric-alarm \
  --alarm-name "SUA_SENHA_BANCO_AQUI-cpu-high" \
  --alarm-description "CPU alta na instância SUA_SENHA_BANCO_AQUI" \
  --metric-name CPUUtilization \
  --namespace AWS/EC2 \
  --statistic Average \
  --period 300 \
  --threshold 85 \
  --comparison-operator GreaterThanThreshold \
  --evaluation-periods 2 \
  --alarm-actions arn:aws:sns:us-east-1:123456789012:SUA_SENHA_BANCO_AQUI-alerts
```

## <IonicIcon name="bug-outline" style={{fontSize: '24px', color: '#ea4b71'}} /> Troubleshooting

### Problemas Comuns

#### Instância não inicia

```bash
# Verificar logs do sistema
aws ec2 get-console-output --instance-id i-1234567890abcdef0

# Verificar status da instância
aws ec2 describe-instances --instance-ids i-1234567890abcdef0
```

#### Problemas de conectividade

```bash
# Testar conectividade com RDS
aws rds describe-db-instances --db-instance-identifier SUA_SENHA_BANCO_AQUI-db

# Verificar security groups
aws ec2 describe-security-groups --group-ids sg-1234567890abcdef0
```
